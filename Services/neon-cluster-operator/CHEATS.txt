# neon delete crd containerregistries.neonkube.io

set DEBUG_ORG=ghcr.io/neonkube-dev
set DEBUG_REPO=neon-cluster-operator
set DEBUG_TAG=neonkube-0.6.0-alpha

cd %NF_ROOT%\Images\neon-cluster-operator
pwsh -f publish.ps1

cd %NF_ROOT%\Lib\Neon.Kube.Setup\Resources\Helm\neon-cluster-operator

neon helm install neon-cluster-operator --namespace neon-system -f values.yaml --set logLevel=debug --set image.pullPolicy=Always --set image.organization=%DEBUG_ORG% --set image.repository=%DEBUG_REPO% --set image.tag=%DEBUG_TAG% .
